// Server

function Main
{
    start program

    call InitServer

    call ServerLoop

    call CleanupSocket
}

function InitServer
{
    create server socket

    listen for clients
}

function ServerLoop
{
    while true do
    {
        check sockets for event

        if new client connects
        {
            accept client
            add client to ClientList
        }
        for each client do
        {
            if client sent text
            {
                receive text from client

                for each client do
                {
                    if client is either the listen socket or the one who sent the text
                    {
                        skip to next socket
                    }

                    send text to client
                }
            }
        }
    }
}

function CleanupSocket
{
    transmit teardown message to all clients

    teardown server socket
}
